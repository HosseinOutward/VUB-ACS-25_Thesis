\begin{adjustbox}{max width=\linewidth}
\begin{tikzpicture}[x=1.2cm,y=0.42cm]
  %----- styles
  \tikzset{
    lane/.style={line width=0.5pt, draw=black!40},
    gridline/.style={thin, draw=black!15},
    protolabel/.style={anchor=east, font=\scriptsize},
    roundlbl/.style={font=\scriptsize},
    sepbar/.style={draw=black!35, line width=0.6pt}
  }

  %----- marker macros
  \newcommand{\Warmup}[2]{% orange diamond
    \begin{scope}[shift={({#1},{#2})},x=1pt,y=1pt]
      \path[draw=orange,fill=orange] (0,2) -- (2,0) -- (0,-2) -- (-2,0) -- cycle;
    \end{scope}}
  \newcommand{\MTQ}[2]{% green circle
    \begin{scope}[shift={({#1},{#2})},x=1pt,y=1pt]
      \path[draw=green!70!black,fill=green!70!black] (0,0) circle (2);
    \end{scope}}
  \newcommand{\TQ}[2]{% blue square
    \begin{scope}[shift={({#1},{#2})},x=1pt,y=1pt]
      \path[draw=blue,fill=blue] (-2,-2) rectangle (2,2);
    \end{scope}}
  \newcommand{\DotsRound}[1]{% vertical "skipped rounds" dots spanning all protocols
    \foreach \yy in {1.5,6.5,11.5} {
      \node[font=\large, text=black!45] at (#1,\yy) {$\cdots$};
    }
  }

  %----- parameters
  \def\nlines{5}        % lanes per protocol
  \def\blockstep{5}     % vertical spacing between protocol blocks
  \def\yTop{15}         % top for gridlines (fits 3 blocks of 5 lanes)

  %----- vertical round gridlines: boundaries 0..11 (=> 11 rounds)
  \foreach \X in {0,...,11} {
    \draw[gridline] (\X,-0.8) -- (\X,\yTop);
  }

  %----- phase labels per round (colored)
  % R1 Warmup; R2 MT; R3 MT (P2,P3); R4-5 Temporal (P3); R6 Empty;
  % R7 … (dots); R8 Empty; R9 MT (P2,P3); R10-11 Temporal (P3)
  \foreach \xc/\txt/\col in {
      0.5/Warmup/orange,
      1.5/Multi-terminal/green!70!black,
      2.5/Multi-terminal/green!70!black,
      3.5/Temporal/blue,
      4.5/Temporal/blue,
      5.5/Empty/black!55,
      6.5/\(\ldots\)/black!55,
      7.5/Empty/black!55,
      8.5/Multi-terminal/green!70!black,
      9.5/Temporal/blue,
      10.5/Temporal/blue
  }{
    \node[roundlbl, text=\col] at (\xc,-1.0) {\txt};
  }

  %----- protocol blocks (3 protocols x 5 lanes)
  \foreach \p [count=\pi from 1] in {Protocol 1,Protocol 2,Protocol 3} {
    \pgfmathsetmacro{\ybase}{(\pi-1)*\blockstep}

    % label the protocol
    \node[protolabel] at (-0.15, {\ybase+2}) {\p};

    % lanes
    \foreach \l in {0,...,4} {
      \pgfmathsetmacro{\yline}{\ybase+\l}
      \draw[lane] (0,\yline) -- (11,\yline);

      % R1: Warmup (all)
      \Warmup{0.95}{\yline}

      % R2: MT (all protocols) — sequential within round
      \pgfmathsetmacro{\xgA}{1.08 + 0.12*\l}
      \MTQ{\xgA}{\yline}

      % R3: MT (Protocols 2 & 3)
      \ifnum\pi>1
        \pgfmathsetmacro{\xgB}{2.08 + 0.12*\l}
        \MTQ{\xgB}{\yline}
      \fi

      % R4–R5: Temporal (Protocol 3, simultaneous at round start)
      \ifnum\pi=3
        \TQ{3.05}{\yline} % Round 4
        \TQ{4.05}{\yline} % Round 5
      \fi

      % R8: Empty (no events)

      % R9: MT (Protocols 2 & 3)
      \ifnum\pi>1
        \pgfmathsetmacro{\xgC}{8.08 + 0.12*\l}
        \MTQ{\xgC}{\yline}
      \fi

      % R10–R11: Temporal (Protocol 3)
      \ifnum\pi=3
        \TQ{9.05}{\yline}  % Round 10
        \TQ{10.05}{\yline} % Round 11
      \fi
    }
  }

  %----- "skipped rounds" dots (once for all protocols)
  \DotsRound{6.5}

  %----- horizontal separators between protocol blocks
  \draw[sepbar] (0,4.5) -- (11,4.5);
  \draw[sepbar] (0,9.5) -- (11,9.5);

  %----- compact legend (top-right)
  \begin{scope}[shift={(8.4,13.7)}]
    \draw[fill=white, draw=black!30, rounded corners=2pt] (-0.25,-1.2) rectangle (2.9,0.55);
    \Warmup{0}{0}         \node[anchor=west, font=\scriptsize] at (0.30,0) {Warmup};
    \MTQ{0}{-0.45}        \node[anchor=west, font=\scriptsize] at (0.30,-0.45) {Multi-terminal quantization};
    \TQ{0}{-0.90}         \node[anchor=west, font=\scriptsize] at (0.30,-0.90) {Temporal-only quantization};
  \end{scope}

  % axis labels (small)
  \node[font=\scriptsize] at (5.5,-1.6) {Rounds};
  \node[rotate=90, font=\scriptsize] at (-0.9,7.5) {Protocols};

\end{tikzpicture}
\end{adjustbox}
