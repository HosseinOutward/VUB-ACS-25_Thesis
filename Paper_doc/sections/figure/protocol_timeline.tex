\begin{adjustbox}{max width=\linewidth}
\begin{tikzpicture}[x=1.0cm,y=0.40cm]
  %----- styles
  \tikzset{
    lane/.style={line width=0.45pt, draw=black!40},
    gridline/.style={thin, draw=black!15},
    roundlbl/.style={font=\scriptsize},
    ticklbl/.style={font=\scriptsize, anchor=east}
  }

  %----- marker macros
  \newcommand{\Init}[2]{\begin{scope}[shift={({#1},{#2})},x=1pt,y=1pt]
    \path[draw=red,fill=red] (0,2.4) -- (2.4,-2.4) -- (-2.4,-2.4) -- cycle;
  \end{scope}}
  \newcommand{\Warmup}[2]{\begin{scope}[shift={({#1},{#2})},x=1pt,y=1pt]
    \path[draw=orange,fill=orange] (0,2) -- (2,0) -- (0,-2) -- (-2,0) -- cycle;
  \end{scope}}
  \newcommand{\MTQ}[2]{\begin{scope}[shift={({#1},{#2})},x=1pt,y=1pt]
    \path[draw=green!70!black,fill=green!70!black] (0,0) circle (2);
  \end{scope}}
  \newcommand{\TQ}[2]{\begin{scope}[shift={({#1},{#2})},x=1pt,y=1pt]
    \path[draw=blue,fill=blue] (-2,-2) rectangle (2,2);
  \end{scope}}
  \newcommand{\DotsCol}[1]{\foreach \yy in {0.5,2.0,3.5} {\node[font=\large, text=black!45] at (#1,\yy) {$\cdots$};}}

  %----- parameters
  \def\xmax{11}
  \def\yTop{4.2}

  %----- vertical gridlines
  \foreach \X in {0,...,\xmax} {
    \draw[gridline] (\X,-0.8) -- (\X,\yTop);
  }

  %----- tilted phase labels per round
  \foreach \xc/\txt/\col in {
      0.5/Initialization/red,
      1.5/Warmup/orange,
      2.5/Warmup/orange,
      3.5/Temporal Quantization/blue,
      4.5/Temporal Quantization/blue,
      5.5/No training (freeze)/black!55,
      6.5/\(\ldots\)/black!55,
      7.5/No training (freeze)/black!55,
      8.5/Temporal Quantization/blue,
      9.5/Temporal Quantization/blue,
      10.5/All-SI Quantization/green!70!black,
      11.5/\(\ldots\)/black!55
  }{
    \node[roundlbl, text=\col, rotate=35, anchor=east] at (\xc,-1.05) {\txt};
  }

  %----- lanes (5 agents)
  \foreach \l in {0,...,4} {
    \pgfmathsetmacro{\yline}{\l}
    \draw[lane] (0,\yline) -- (\xmax,\yline);

    % R1 Initialization (end of round)
    \Init{0.95}{\yline}

    % R2–R3 Warmup (sequential)
    \pgfmathsetmacro{\xwA}{1.08 + 0.14*\l}
    \pgfmathsetmacro{\xwB}{2.08 + 0.14*\l}
    \Warmup{\xwA}{\yline}
    \Warmup{\xwB}{\yline}

    % R4–R5 Temporal (simultaneous at start)
    \TQ{3.05}{\yline}
    \TQ{4.05}{\yline}

    % R9–R10 Temporal (simultaneous)
    \TQ{8.05}{\yline}
    \TQ{9.05}{\yline}

    % R11 All-SI (simultaneous)
    \MTQ{10.05}{\yline}
  }

  % skipped rounds dots at Round 7
  \DotsCol{6.5}

  % Agents axis (y)
  \draw[black!40] (-0.06,0) -- (-0.06,4);
  \foreach \l in {0,...,4} {
    \draw[black!40] (-0.09,\l) -- (-0.03,\l);
    \node[ticklbl] at (-0.12,\l) {\scriptsize \the\numexpr\l+1\relax};
  }
  \node[rotate=90, font=\scriptsize] at (-0.5,2) {Agents};

  % Legend (raised, no box, looser spacing)
  \begin{scope}[shift={(5.5,5.3)}] % push up a bit
    % left column
    \Init{-3.2}{0.30}     \node[anchor=west, font=\scriptsize] at (-2.8,0.30) {Initial Training};
    \Warmup{-3.2}{-0.60}  \node[anchor=west, font=\scriptsize] at (-2.8,-0.60) {Warmup Training};
    % right column
    \TQ{0.1}{0.30}        \node[anchor=west, font=\scriptsize] at (0.45,0.30) {Temporal Quantization Training};
    \MTQ{0.1}{-0.60}      \node[anchor=west, font=\scriptsize] at (0.45,-0.60) {All-SI Quantization Training};
  \end{scope}

  % Rounds axis label (lower)
  \node[font=\scriptsize] at (5.5,-3.0) {Rounds};
